<span class="s0">def </span><span class="s1">create_overlap_set(constant_names):</span>
    <span class="s4">&quot;&quot;&quot;Helper function to find overlapping constant values/names. 
 
    Returns a set of fronzensets: 
        set(frozenset(names of overlapping constants), ...) 
    &quot;&quot;&quot;</span>
    <span class="s2"># Create an overlap dict.</span>
    <span class="s1">overlap_dict = {}</span>

    <span class="s0">for </span><span class="s1">name </span><span class="s0">in </span><span class="s1">constant_names:</span>
        <span class="s1">value = getattr(pygame.constants</span><span class="s0">, </span><span class="s1">name)</span>
        <span class="s1">overlap_dict.setdefault(value</span><span class="s0">, </span><span class="s1">set()).add(name)</span>

    <span class="s2"># Get all entries with more than 1 value.</span>
    <span class="s1">overlaps = set()</span>

    <span class="s0">for </span><span class="s1">overlap_names </span><span class="s0">in </span><span class="s1">overlap_dict.values():</span>
        <span class="s0">if </span><span class="s1">len(overlap_names) &gt; </span><span class="s5">1</span><span class="s1">:</span>
            <span class="s1">overlaps.add(frozenset(overlap_names))</span>

    <span class="s0">return </span><span class="s1">overlaps</span>

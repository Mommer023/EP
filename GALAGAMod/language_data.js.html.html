  <span class="s2">var </span><span class="s1">C = c + </span><span class="s3">&quot;[^aeiouy]*&quot;</span><span class="s1">;    </span><span class="s0">// consonant sequence</span>

  <span class="s2">var </span><span class="s1">mgr0 = </span><span class="s3">&quot;^(&quot; </span><span class="s1">+ C + </span><span class="s3">&quot;)?&quot; </span><span class="s1">+ V + C;                      </span><span class="s0">// [C]VC... is m&gt;0</span>

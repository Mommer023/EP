<span class="s0">/**

<span class="s1">const _highlight = (node, addItems, text, className) =&gt; {</span>

      <span class="s1">!parent.classList.contains(</span><span class="s2">&quot;nohighlight&quot;</span><span class="s1">)</span>

      <span class="s1">}</span>

      <span class="s1">span.appendChild(document.createTextNode(val.substr(pos, text.length)));</span>
      <span class="s1">parent.insertBefore(</span>
        <span class="s1">span,</span>
        <span class="s1">parent.insertBefore(</span>

          <span class="s1">node.nextSibling</span>
        <span class="s1">)</span>
      <span class="s1">);</span>
      <span class="s1">node.nodeValue = val.substr(</span><span class="s4">0</span><span class="s1">, pos);</span>

        <span class="s1">const rect = document.createElementNS(</span>
          <span class="s2">&quot;http://www.w3.org/2000/svg&quot;</span><span class="s1">,</span>
          <span class="s2">&quot;rect&quot;</span>

        <span class="s1">rect.x.baseVal.value = bbox.x;</span>

        <span class="s1">rect.width.baseVal.value = bbox.width;</span>
        <span class="s1">rect.height.baseVal.value = bbox.height;</span>

      <span class="s1">}</span>
    <span class="s1">}</span>
  <span class="s1">} </span><span class="s3">else if </span><span class="s1">(node.matches &amp;&amp; !node.matches(</span><span class="s2">&quot;button, select, textarea&quot;</span><span class="s1">)) {</span>
    <span class="s1">node.childNodes.forEach((el) =&gt; _highlight(el, addItems, text, className));</span>
  <span class="s1">}</span>
<span class="s1">};</span>
